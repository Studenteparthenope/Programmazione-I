typedef struct{ //creazione della struct con i campi parola,significato,e sinonimi
char parola [100];
char significato[100];
char *sinonimi[5];
}Parola;
//Prototipi
void opzioni();
void visualizza_significato(char *);
void creazione_dizionario();
Parola Dizionario[20];
int parole_dizionario=5;
//main del programma
int main(){
creazione_dizionario();
ordinamento();
opzioni();
return 0;
}
//function per la scelta nel menù principale
void opzioni(){
char scelta;
char parola[30];
while(1){
printf("[1]Inserire parola\n");
printf("[2]Visualizza il significato\n");
printf("[3]Exit");
printf("\n");
scelta = getch();
switch(scelta){
case '1':
Inserimento_parola();
break;
case '2':
printf("\nInserisci la parola da ricercare: ");
scanf("%s",&parola);
fflush(stdin);
visualizza_significato(parola);
printf("\n");
system("pause");
break;
}
if(scelta == '3')
break;
system("cls");
}
}
//function per la visualizzazione del significato delle parole presenti nel dizionario
void visualizza_significato(char parola[]){
int i;
for(i=0;i<parole_dizionario;i++){ //scorrimento del dizionario
if(strcmp(Dizionario[i].parola,parola) == 0){ //verifica di uguaglianza
printf("\n%s",Dizionario[i].significato); //in caso di uguaglianza stampa del corrispondente significato
return;
}
}
printf("\nLa parola non si trova nel Dizionario!\n"); //stampa in causa di mancata uguaglianza
}
//function per inserimento delle parole nel dizionario
void Inserimento_parola(){
char parola[100];
char significato[100];
printf("\nDigita la parola da inserire: ");
gets(parola);
fflush(stdin);
printf("\nInserire il suo significato: ");
gets(significato);
fflush(stdin);
strcpy(Dizionario[parole_dizionario].parola,parola);
strcpy(Dizionario[parole_dizionario].significato,significato);
parole_dizionario++;
ordinamento();
}
//function per l'ordinamento delle parole del dizioanrio
void ordinamento(){
int i,j;
Parola el_da_ins;
for(i=1;i<parole_dizionario;i++){
el_da_ins=Dizionario[i];
j=i-1;
while(j>0 && strcmp(el_da_ins.parola,Dizionario[j].parola)<0){
Dizionario[j+1]=Dizionario[j];
j--;
}
Dizionario[j+1]=el_da_ins;
}
}
